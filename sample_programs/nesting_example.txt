declare printf(_ : string, ...) : integer;

fun main( a : integer ) : integer = (
    { a = 5 },
    { b = true },
    printf('%d, %d\n', a, b),
    notranja('tukaj'),
    0
) { where var b : logical;
    fun notranja ( c : string ) : integer = (
        printf('%d, %d, %s\n', a, b, c)

        printf('%d\n',a)
    )}
}
