declare printf(_ : string, ...) : integer;

fun power(x : integer, n : integer) : integer = (
    { z = 1 },
    { if ( n == 0 ) then { z = 1 }
    else { for i = 0, n, 1 : { z = z * x }}},
    z) { where var z : integer; var i : integer };

fun main( argc : integer ) : integer = (
    { x = 5 }, { n = 4 },
    {x = power(x, n)},
    printf('%d\n', x)
) { where var n : integer; var x : integer}
